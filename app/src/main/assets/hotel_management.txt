int OPERATE::no_day(void)
{

getdate(&d);
int a[13]={0,31,29,31,30,31,30,31,31,30,31,30,31};
if(d.da_mon==cust.dt.da_mon)
n=d.da_day-cust.dt.da_day+1;
else
{
for(int i=cust.dt.da_mon;i<=d.da_mon;++i)
{
if(i==cust.dt.da_mon)
n+=(a[i]-cust.dt.da_day+1);
else
{
if(i==d.da_mon)
n+=d.da_day;
else
n+=a[i];
}
}
}
return n;
}
